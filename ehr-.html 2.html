    function searchPatients(query) {

      const patientList = document.getElementById('patient-list');

     

      if (!query.trim()) {

        renderPatientList();

        return;

      }

 

      const filteredPatients = ehrData.patients.filter(patient => {

        const fullName = `${patient.firstName} ${patient.lastName}`.toLowerCase();

        const searchTerm = query.toLowerCase();

       

        return fullName.includes(searchTerm) || patient.phone.includes(searchTerm);

      });

 

      if (filteredPatients.length === 0) {

        patientList.innerHTML = `

          <div class="text-center" style="padding: 40px;">

            <p class="text-muted">No patients found matching "${query}"</p>

          </div>

        `;

        return;

      }

 

      patientList.innerHTML = filteredPatients.map(patient => {

        const age = calculateAge(patient.dob);

        const lastVisit = getLastVisitDate(patient.id);

        const visitCount = ehrData.visits.filter(v => v.patientId === patient.id).length;

       

        return `

          <div class="patient-card" onclick="showPatientDetails(${patient.id})">

            <div class="patient-name">${patient.firstName} ${patient.lastName}</div>

            <div class="patient-info">

              <span>Age: ${age}</span>

              <span>Sex: ${patient.sex}</span>

    /**

     * Calculate age from date of birth

     */

    function calculateAge(dob) {

      const today = new Date();

      const birthDate = new Date(dob);

      let age = today.getFullYear() - birthDate.getFullYear();

      const monthDiff = today.getMonth() - birthDate.getMonth();

     

      if (monthDiff < 0 || (monthDiff === 0 && today.getDate() < birthDate.getDate())) {

        age--;

      }

     

      return age;

    }

    /**

     * Calculate age from date of birth

     */

    function calculateAge(dob) {

      const today = new Date();

      const birthDate = new Date(dob);

      let age = today.getFullYear() - birthDate.getFullYear();

      const monthDiff = today.getMonth() - birthDate.getMonth();

     

      if (monthDiff < 0 || (monthDiff === 0 && today.getDate() < birthDate.getDate())) {

        age--;

      }

     

      return age;

    }
